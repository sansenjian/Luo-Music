# 本地音乐功能完整技术方案（生产级）

## 1. 设计目标与边界

### 1.1 功能边界

- **文件夹管理**：添加、移除、启用/禁用本地音乐文件夹。
- **文件扫描**：递归扫描，支持 MP3、FLAC、M4A、OGG、WAV、AAC、APE 等主流格式。
- **元数据解析**：提取标题、艺术家、专辑、封面、时长等，封面以文件形式存储，仅哈希入库。
- **增量更新**：支持全量扫描、基于 `mtime` 的增量扫描，以及文件系统实时监听。
- **媒体库持久化**：SQLite 存储所有歌曲信息，支持按艺术家、专辑等维度查询。
- **UI 展示**：提供歌曲列表、艺术家/专辑视图，支持虚拟滚动。
- **与播放器联动**：双击播放，自动记录到“最近播放”模块。
- **性能目标**：可流畅管理 10 万+ 歌曲，扫描 1000 首歌耗时 < 30 秒，界面始终保持 60fps。

**数据库封装（`better-sqlite3-pool`）**：

```typescript
// main/database/libraryDB.ts
import { Database } from 'better-sqlite3-pool';
import path from 'path';
import { app } from 'electron';
import { v4 as uuidv4 } from 'uuid';

export class LibraryDatabase {
  private dbPromise: Promise<Database>;

  constructor() {
    const dbPath = path.join(app.getPath('userData'), 'library.db');
    this.dbPromise = Database.create(dbPath, {
      minWorkers: 1,
      maxWorkers: 2,   // 主线程 + 1 个写入线程，过多会增加锁竞争
    });
    this.init();
  }

  private async init() {
    const db = await this.dbPromise;
    await db.pragma('journal_mode = WAL');
    await db.pragma('busy_timeout = 5000');
    await db.exec(`
      CREATE TABLE IF NOT EXISTS folders (...);
      CREATE TABLE IF NOT EXISTS songs (...);
      CREATE INDEX IF NOT EXISTS ...
    `);
  }

  // 文件夹管理
  async addFolder(folderPath: string): Promise<string> {
    const db = await this.dbPromise;
    const id = uuidv4();
    await db.run(`INSERT INTO folders (id, path, created_at) VALUES (?, ?, ?)`, [id, folderPath, Date.now()]);
    return id;
  }

  async getFolders(): Promise<Folder[]> {
    const db = await this.dbPromise;
    return db.all(`SELECT * FROM folders WHERE enabled = 1`);
  }

  async removeFolder(id: string): Promise<void> {
    const db = await this.dbPromise;
    await db.run(`DELETE FROM folders WHERE id = ?`, [id]);
  }

  // 歌曲增删改查
  async upsertSong(song: SongData): Promise<void> {
    const db = await this.dbPromise;
    await db.run(`
      INSERT INTO songs (
        id, file_path, folder_id, title, artist, album, album_artist,
        track_no, year, genre, duration, bitrate, sample_rate, channels,
        cover_hash, file_size, file_mtime, format, created_at, updated_at
      ) VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)
      ON CONFLICT(file_path) DO UPDATE SET
        title = excluded.title,
        artist = excluded.artist,
        album = excluded.album,
        cover_hash = excluded.cover_hash,
        file_mtime = excluded.file_mtime,
        updated_at = excluded.updated_at
    `, [ ... ]);
  }

  // 游标分页查询（按 updated_at DESC）
  async getSongsByCursor(limit: number, cursor?: number): Promise<{ songs: SongData[], nextCursor: number | null }> {
    const db = await this.dbPromise;
    let sql = `SELECT * FROM songs `;
    const params: any[] = [];
    if (cursor) {
      sql += `WHERE updated_at < ? `;
      params.push(cursor);
    }
    sql += `ORDER BY updated_at DESC LIMIT ?`;
    params.push(limit);

    const songs = await db.all(sql, params);
    const nextCursor = songs.length === limit ? songs[songs.length - 1].updated_at : null;
    return { songs, nextCursor };
  }

  async getAllCoverHashes(): Promise<string[]> {
    const db = await this.dbPromise;
    const rows = await db.all(`SELECT DISTINCT cover_hash FROM songs WHERE cover_hash IS NOT NULL`);
    return rows.map((r: any) => r.cover_hash);
  }

  async deleteSongsByFilePaths(filePaths: string[]): Promise<void> {
    if (filePaths.length === 0) return;
    const db = await this.dbPromise;
    const placeholders = filePaths.map(() => '?').join(',');
    await db.run(`DELETE FROM songs WHERE file_path IN (${placeholders})`, filePaths);
  }

  async transaction<T>(fn: (db: Database) => Promise<T>): Promise<T> {
    const db = await this.dbPromise;
    return db.transaction(fn);
  }
}
```

### 3.2 封面管理器（CoverManager）

文件存储路径：

- 原始图：`userData/covers/originals/{hash}.jpg`
- 缩略图：`userData/covers/thumbnails/{hash}_thumb.jpg`

```typescript
// main/services/CoverManager.ts
import fs from 'fs'
import path from 'path'
import crypto from 'crypto'
import sharp from 'sharp'
import { app } from 'electron'
import { parseFile } from 'music-metadata'

const THUMBNAIL_SIZE = 120
const THUMBNAIL_QUALITY = 80
const ORIGINAL_MAX_SIZE = 1024

export class CoverManager {
  private originalsDir: string
  private thumbnailsDir: string

  constructor() {
    const coversDir = path.join(app.getPath('userData'), 'covers')
    this.originalsDir = path.join(coversDir, 'originals')
    this.thumbnailsDir = path.join(coversDir, 'thumbnails')
    ;[this.originalsDir, this.thumbnailsDir].forEach(dir => {
      if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true })
    })
  }

  private hashBuffer(buffer: Buffer): string {
    return crypto.createHash('sha256').update(buffer).digest('hex')
  }

  async extractAndSaveCover(filePath: string): Promise<string | null> {
    try {
      const metadata = await parseFile(filePath)
      const picture = metadata.common.picture?.[0]
      if (!picture) return null

      const hash = this.hashBuffer(picture.data)
      const originalPath = path.join(this.originalsDir, `${hash}.jpg`)
      const thumbnailPath = path.join(this.thumbnailsDir, `${hash}_thumb.jpg`)

      if (fs.existsSync(originalPath) && fs.existsSync(thumbnailPath)) {
        return hash
      }

      const image = sharp(picture.data)
      const meta = await image.metadata()

      // 保存原始图（限制尺寸）
      let pipeline = image
      if (meta.width && meta.width > ORIGINAL_MAX_SIZE) {
        pipeline = pipeline.resize(ORIGINAL_MAX_SIZE, null, { withoutEnlargement: true })
      }
      await pipeline.jpeg({ quality: 85 }).toFile(originalPath)

      // 保存缩略图
      await image
        .resize(THUMBNAIL_SIZE, THUMBNAIL_SIZE, { fit: 'cover' })
        .jpeg({ quality: THUMBNAIL_QUALITY })
        .toFile(thumbnailPath)

      return hash
    } catch {
      return null
    }
  }

  getThumbnailBase64(hash: string): string | null {
    try {
      const p = path.join(this.thumbnailsDir, `${hash}_thumb.jpg`)
      if (!fs.existsSync(p)) return null
      const buffer = fs.readFileSync(p)
      return `data:image/jpeg;base64,${buffer.toString('base64')}`
    } catch {
      return null
    }
  }

  getOriginalBase64(hash: string): string | null {
    try {
      const p = path.join(this.originalsDir, `${hash}.jpg`)
      if (!fs.existsSync(p)) return null
      const buffer = fs.readFileSync(p)
      return `data:image/jpeg;base64,${buffer.toString('base64')}`
    } catch {
      return null
    }
  }

  async cleanupOrphanCovers(validHashes: Set<string>): Promise<void> {
    // 使用 Worker 异步清理（见 4.5 节）
  }
}
```

```typescript
// main/scanner.ts
import Piscina from 'piscina'
import fs from 'fs'
import path from 'path'
import { v4 as uuidv4 } from 'uuid'
import { libraryDB } from './database/libraryDB'
import { CoverManager } from './services/CoverManager'

const SUPPORTED_EXT = new Set(['.mp3', '.flac', '.m4a', '.ogg', '.wav', '.aac', '.ape'])

export class LibraryScanner {
  private pool: Piscina
  private coverManager: CoverManager
  private abortControllers: Map<string, AbortController> = new Map()

  constructor() {
    this.coverManager = new CoverManager()
    this.pool = new Piscina({
      filename: path.join(__dirname, '../workers/metadata-worker.js'),
      maxThreads: 4,
      idleTimeout: 30000
    })
  }

  private collectAudioFiles(dirPath: string): string[] {
    const results: string[] = []
    const traverse = (current: string) => {
      const entries = fs.readdirSync(current, { withFileTypes: true })
      for (const e of entries) {
        const full = path.join(current, e.name)
        if (e.isDirectory()) traverse(full)
        else if (e.isFile() && SUPPORTED_EXT.has(path.extname(e.name).toLowerCase())) {
          results.push(full)
        }
      }
    }
    traverse(dirPath)
    return results
  }

  private async processFile(filePath: string, folderId: string): Promise<SongData> {
    const stats = fs.statSync(filePath)
    const result = await this.pool.run(filePath)
    if (!result.success) throw new Error(result.error)
    const metadata = result.metadata

    const coverHash = await this.coverManager.extractAndSaveCover(filePath)

    const { common, format } = metadata
    return {
      id: uuidv4(),
      file_path: filePath,
      folder_id: folderId,
      title: common.title || path.basename(filePath, path.extname(filePath)),
      artist: common.artist || null,
      album: common.album || null,
      album_artist: common.albumartist || null,
      track_no: common.track?.no || null,
      year: common.year || null,
      genre: common.genre?.[0] || null,
      duration: Math.floor(format.duration || 0),
      bitrate: format.bitrate || null,
      sample_rate: format.sampleRate || null,
      channels: format.numberOfChannels || null,
      cover_hash: coverHash,
      file_size: stats.size,
      file_mtime: stats.mtimeMs,
      format: format.container || path.extname(filePath).slice(1),
      created_at: Date.now(),
      updated_at: Date.now()
    }
  }

  async scanFolder(
    folderId: string,
    folderPath: string,
    onProgress?: (p: ScanProgress) => void
  ): Promise<void> {
    const controller = new AbortController()
    this.abortControllers.set(folderId, controller)

    const files = this.collectAudioFiles(folderPath)
    const progress: ScanProgress = {
      folderId,
      totalFiles: files.length,
      processedFiles: 0,
      currentFile: '',
      status: 'scanning'
    }
    onProgress?.(progress)

    const BATCH_SIZE = 20 // 每批处理 20 个，平衡并发和内存
    for (let i = 0; i < files.length; i += BATCH_SIZE) {
      if (controller.signal.aborted) break

      const batch = files.slice(i, i + BATCH_SIZE)
      const promises = batch.map(file =>
        this.processFile(file, folderId).catch(e => {
          console.error(`处理失败: ${file}`, e)
          return null
        })
      )

      const results = await Promise.all(promises)

      // 批量事务写入
      await libraryDB.transaction(async db => {
        for (const song of results) {
          if (song) await db.upsertSong(song)
        }
      })

      progress.processedFiles += batch.length
      onProgress?.(progress)
    }

    progress.status = controller.signal.aborted ? 'paused' : 'completed'
    await libraryDB.updateFolderLastScan(folderId)
    onProgress?.(progress)
    this.abortControllers.delete(folderId)
  }

  stopScan(folderId: string) {
    this.abortControllers.get(folderId)?.abort()
  }

  // 增量扫描类似，省略
}
```

### 3.4 文件监听器（防抖批量合并）

```typescript
// main/watcher.ts
import chokidar from 'chokidar'
import { debounce } from 'lodash-es'
import { libraryDB } from './database/libraryDB'
import { LibraryScanner } from './scanner'
import { BrowserWindow } from 'electron'

export class LibraryWatcher {
  private watchers: Map<string, chokidar.FSWatcher> = new Map()
  private pendingAdds: Set<string> = new Set()
  private pendingChanges: Set<string> = new Set()
  private pendingUnlinks: Set<string> = new Set()
  private folderIdMap: Map<string, string> = new Map() // filePath -> folderId

  constructor(private scanner: LibraryScanner) {}

  private debouncedProcess = debounce(() => this.processPendingQueue(), 2000)

  startWatching(folderId: string, folderPath: string) {
    if (this.watchers.has(folderId)) return

    const watcher = chokidar.watch(folderPath, {
      ignored: /(^|[\/\\])\../,
      persistent: true,
      ignoreInitial: true,
      awaitWriteFinish: { stabilityThreshold: 2000, pollInterval: 100 }
    })

    watcher
      .on('add', filePath => {
        if (this.isAudioFile(filePath)) {
          this.folderIdMap.set(filePath, folderId)
          this.pendingAdds.add(filePath)
          this.debouncedProcess()
        }
      })
      .on('change', filePath => {
        if (this.isAudioFile(filePath)) {
          this.folderIdMap.set(filePath, folderId)
          this.pendingChanges.add(filePath)
          this.debouncedProcess()
        }
      })
      .on('unlink', filePath => {
        if (this.isAudioFile(filePath)) {
          this.pendingUnlinks.add(filePath)
          this.debouncedProcess()
        }
      })

    this.watchers.set(folderId, watcher)
  }

  stopWatching(folderId: string) {
    this.watchers.get(folderId)?.close()
    this.watchers.delete(folderId)
  }

  private async processPendingQueue() {
    const adds = Array.from(this.pendingAdds)
    const changes = Array.from(this.pendingChanges)
    const unlinks = Array.from(this.pendingUnlinks)
    this.pendingAdds.clear()
    this.pendingChanges.clear()
    this.pendingUnlinks.clear()

    // 处理删除
    if (unlinks.length > 0) {
      await libraryDB.deleteSongsByFilePaths(unlinks)
    }

    // 处理新增和修改（合并解析）
    const toProcess = [...adds, ...changes]
    if (toProcess.length > 0) {
      const results = await Promise.allSettled(
        toProcess.map(filePath => {
          const folderId = this.folderIdMap.get(filePath)!
          return this.scanner.processFile(filePath, folderId)
        })
      )

      await libraryDB.transaction(async db => {
        for (const r of results) {
          if (r.status === 'fulfilled' && r.value) {
            await db.upsertSong(r.value)
          }
        }
      })
    }

    this.broadcastUpdate()
  }

  private isAudioFile(filePath: string): boolean {
    const ext = path.extname(filePath).toLowerCase()
    return ['.mp3', '.flac', '.m4a', '.ogg', '.wav', '.aac'].includes(ext)
  }

  private broadcastUpdate() {
    BrowserWindow.getAllWindows().forEach(win => {
      win.webContents.send('library:updated')
    })
  }
}
```

```typescript
// workers/cover-cleaner-worker.ts
import fs from 'fs'
import path from 'path'
import { parentPort } from 'worker_threads'

parentPort?.on('message', ({ originalsDir, thumbnailsDir, validHashes }) => {
  const validSet = new Set(validHashes)

  const cleanDir = (dir: string) => {
    if (!fs.existsSync(dir)) return
    const files = fs.readdirSync(dir)
    for (const file of files) {
      const hash = file.replace(/_thumb\.jpg$/, '').replace(/\.jpg$/, '')
      if (!validSet.has(hash)) {
        try {
          fs.unlinkSync(path.join(dir, file))
        } catch {}
      }
    }
  }

  cleanDir(originalsDir)
  cleanDir(thumbnailsDir)
  parentPort?.postMessage({ done: true })
})
```

```typescript
    getFolders: () => Promise<Folder[]>
    removeFolder: (id: string) => Promise<boolean>
    scanFolder: (id: string, path: string) => Promise<{ started: boolean }>
    stopScan: (id: string) => Promise<boolean>
    onScanProgress: (cb: (p: ScanProgress) => void) => void
    getSongsCursor: (
      limit: number,
      cursor?: number
    ) => Promise<{ songs: SongData[]; nextCursor: number | null }>
    getSongCount: () => Promise<number>
    getThumbnail: (hash: string) => Promise<string | null>
    onLibraryUpdated: (cb: () => void) => void
    cleanupOrphanCovers: () => Promise<void>
  }

  interface Window {
    electronAPI: { library: LibraryAPI }
  }
}
```

### 4.2 Pinia Store：音乐库与封面缓存

**音乐库 Store** (`stores/library.ts`)：

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { useRecentStore } from './recent'

export const useLibraryStore = defineStore('library', () => {
  const folders = ref<Folder[]>([])
  const songs = ref<SongData[]>([])
  const totalCount = ref(0)
  const nextCursor = ref<number | null>(null)
  const hasMore = ref(true)
  const isLoading = ref(false)
  const scanProgress = ref<Map<string, ScanProgress>>(new Map())

  async function loadFolders() {
    folders.value = await window.electronAPI.library.getFolders()
  }

  async function loadSongs(limit = 100) {
    isLoading.value = true
    try {
      const result = await window.electronAPI.library.getSongsCursor(limit)
      songs.value = result.songs
      nextCursor.value = result.nextCursor
      hasMore.value = result.nextCursor !== null
      totalCount.value = await window.electronAPI.library.getSongCount()
    } finally {
      isLoading.value = false
    }
  }

  async function loadMore(limit = 100) {
    if (!hasMore.value || isLoading.value) return
    isLoading.value = true
    try {
      const result = await window.electronAPI.library.getSongsCursor(limit, nextCursor.value)
      songs.value.push(...result.songs)
      nextCursor.value = result.nextCursor
      hasMore.value = result.nextCursor !== null
    } finally {
      isLoading.value = false
    }
  }

  async function addFolder(path: string) {
    const { folderId, path: p } = await window.electronAPI.library.addFolder(path)
    await loadFolders()
    window.electronAPI.library.scanFolder(folderId, p)
  }

  async function removeFolder(id: string) {
    await window.electronAPI.library.removeFolder(id)
    await loadAll()
  }

  function playSong(song: SongData) {
    window.dispatchEvent(
      new CustomEvent('play-media', {
        detail: {
          mediaId: song.id,
          title: song.title,
          artist: song.artist,
          coverHash: song.cover_hash,
          duration: song.duration,
          filePath: song.file_path
        }
      })
    )

    const recentStore = useRecentStore()
    recentStore.addRecentItem({
      mediaId: song.id,
      title: song.title,
      artist: song.artist || '',
      cover: song.cover_hash || '',
      duration: song.duration,
      progress: 0
    })
  }

  function setupListeners() {
    window.electronAPI.library.onScanProgress(p => {
      scanProgress.value.set(p.folderId, p)
      if (p.status === 'completed') loadSongs()
    })
    window.electronAPI.library.onLibraryUpdated(() => loadSongs())
  }

  async function loadAll() {
    await Promise.all([loadFolders(), loadSongs()])
  }

  loadAll().then(setupListeners)

  return {
    folders,
    songs,
    totalCount,
    hasMore,
    isLoading,
    scanProgress,
    addFolder,
    removeFolder,
    loadMore,
    playSong,
    loadSongs
  }
})
```

**封面缓存 Store** (`stores/coverCache.ts`)：

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'

const MAX_CACHE = 200

export const useCoverCacheStore = defineStore('coverCache', () => {
  const cache = ref<Map<string, { base64: string; ts: number }>>(new Map())

  function get(hash: string): string | null {
    const entry = cache.value.get(hash)
    if (entry) {
      entry.ts = Date.now()
      return entry.base64
    }
    return null
  }

  async function fetchThumbnail(hash: string): Promise<string> {
    const cached = get(hash)
    if (cached) return cached
    const base64 = await window.electronAPI.library.getThumbnail(hash)
    if (base64) set(hash, base64)
    return base64 || ''
  }

  function set(hash: string, base64: string) {
    if (cache.value.size >= MAX_CACHE) {
      const sorted = Array.from(cache.value.entries()).sort((a, b) => a[1].ts - b[1].ts)
      sorted.slice(0, Math.floor(MAX_CACHE * 0.2)).forEach(([k]) => cache.value.delete(k))
    }
    cache.value.set(hash, { base64, ts: Date.now() })
  }

  return { get, fetchThumbnail }
})
```

```vue
  if (!props.coverHash || loaded.value) return
  const base64 = await cacheStore.fetchThumbnail(props.coverHash)

    src.value = base64

      }
    },
    { rootMargin: '50px' }
  )

})
```
